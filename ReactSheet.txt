3. POST /api/agents/chat/<session_id>/   (SSE 流式响应)
   Content-Type: text/event-stream
   用途：发送消息并接收流式回复
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// 每个 SSE 块格式：
//   event: <event_type>
//   data: <JSON string>
//
// 事件类型与 data 格式：

event: reasoning
data: "正在分析输入意图并扫描本地权限库..."    // string，RAG 判定阶段的进度提示（可能多条）

event: reasoning
data: "记忆锚点命中 2 条"

event: thinking
data: "用户问的是..."                          // string，模型 CoT 思考流（逐 chunk，拼接展示）

event: content
data: "量子纠缠是两个粒子..."                  // string，正文回复流（逐 chunk，拼接展示）

event: anchor_created
data: {"pattern": "量子纠缠"}                  // G045 function call 创建了新记忆锚点时触发

event: error
data: "LLM 传输层中断: ..."                    // string，发生错误时

event: done
data: "[DONE]"                                 // 流结束标志

// 前端处理示例：
//   const es = new EventSource(...)  ← 或用 fetch + ReadableStream
//   es.addEventListener('content', e => fullText += e.data)
//   es.addEventListener('thinking', e => thinkingText += e.data)
//   es.addEventListener('reasoning', e => reasoningSteps.push(e.data))
//   es.addEventListener('done', () => es.close())

7. POST /api/agents/sessions/init/
   用途：创建新会话
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// Request body:
{
  "preset_id": 1,
  "name": "新会话 2026-03-13",                   // 可选，默认 "新会话 <today>"
  "project_ids": [3, 7],                         // 可选，standard agent 传 [单个id]，G045 传多个
  "session_type": "chat",                        // 可选，默认 "chat"，可传 "code"
  "thinking_level": "auto",                      // 可选，默认 "auto"
  "temperature": 1.0                             // 可选，默认 1.0
}

// Response (201):
{
  "msg": "G045 会话已建立，权限已锁定。",
  "data": {
    "session_id": 12,                            // = Conversation.id，聊天接口用这个
    "agent_session_id": 5,
    "session_name": "新会话 2026-03-13"
  }
}

  "name": "新名称",
  "thinking_level": "high",
  "temperature": 0.6
}

// Response (200): 返回完整的 Conversation 对象（同 GET /conversations/ 的单条格式）

// 注意：thinking_level / temperature 也会在每次 POST /chat/<id>/ 发消息后自动写回，
//       所以大部分时候不需要手动 PATCH，只有「用户主动切换但尚未发消息」时才需要。

9. GET /api/agents/presets/<preset_id>/anchors/snapshot/
   返回：Array<MemoryAnchor>
   用途：G045 个人主页只读展示，高分活跃锚点快照
   筛选：is_active=true AND current_weight >= 0.5，随机排序取前 3
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

[
  {
    "trigger_id": "a3f2c1d0-...",               // UUID，唯一标识
    "pattern": "量子纠缠",                        // 触发关键词
    "essential_note": "用户对量子力学有浓厚兴趣，提及时可深入展开...",
    "current_weight": 0.92,                      // 0.0 ~ 1.0，越高越活跃
    "is_persistent": false                       // true = 恒定锚点（永不衰减），给前端加特殊视效
  },
  {
    "trigger_id": "b7e4a2c1-...",
    "pattern": "深夜",
    "essential_note": "用户在深夜工作时情绪偏向沉静内省，回应节奏放慢...",
    "current_weight": 1.0,
    "is_persistent": true                        // ← 满权两周后自动晋升，前端可加光晕/金边等效果
  }
]

// is_persistent 视效建议：
//   false → 普通卡片样式
//   true  → 金色边框 / 发光效果 / 特殊徽章
