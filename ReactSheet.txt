3. POST /api/agents/chat/<session_id>/   (SSE 流式响应)
   Content-Type: text/event-stream
   用途：发送消息并接收流式回复
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// ── 请求体（multipart/form-data 或 JSON）────────────────
// content              string   用户消息正文（regenerate_message_id 模式下可省略）
// thinking_level       string   "off"|"auto"|"low"|"medium"|"high"，默认 "medium"
// temperature          float    0.0~2.0，默认 1.0
// model                string?  模型名称覆盖（可省，使用 preset 默认值）
// pending_attachments  array?   本地文件 pending 元数据列表
// files                File[]?  直接上传的图片/文件（multipart）
//
// ── 消息编辑 / 重生成（三者互斥）────────────────────────
// regenerate           bool     重生成最后一条 AI 消息（旧接口，已修复：会真正删除旧 AI 消息）
// edit_message_id      int?     覆盖指定用户消息（id）并重新生成；后端自动删除该消息之后的所有消息
// regenerate_message_id int?    重新生成指定 AI 消息（id）；后端自动删除该消息及其之后的所有消息
//
// 前端处理逻辑：
//   普通发送：{ content: "..." }
//   编辑用户消息并重发：{ content: "修改后的内容", edit_message_id: 101 }
//     → 后端覆盖消息 101 内容，删除 index > 101.index 的所有消息，重新生成 AI 回复
//   重生成指定 AI 消息：{ regenerate_message_id: 102 }
//     → 后端删除消息 102 及之后的所有消息，基于前驱 user 消息重新生成
//   重生成最后一条：{ regenerate: true, content: "<上一条用户消息>" }
//     （推荐改用 regenerate_message_id 精确指定）
//
// 注意：
//   - edit_message_id / regenerate_message_id 触发后，前端应清除本地对应位置之后的所有消息
//   - Stop（中止流式）：使用 AbortController.abort()，后端无消息入库（持久化在流结束后）

// 每个 SSE 块格式：
//   event: <event_type>
//   data: <JSON string>
//
// 事件类型与 data 格式：

event: reasoning
data: "正在分析输入意图并扫描本地权限库..."    // string，RAG 判定阶段的进度提示（可能多条）

event: reasoning
data: "记忆锚点命中 2 条"

event: thinking
data: "用户问的是..."                          // string，模型 CoT 思考流（逐 chunk，拼接展示）

event: content
data: "量子纠缠是两个粒子..."                  // string，正文回复流（逐 chunk，拼接展示）

event: anchor_created
data: {"pattern": "量子纠缠"}                  // G045 function call 创建了新记忆锚点时触发

event: status
data: "正在搜索…"                              // string，工具调用前的状态提示（阻塞操作期间显示）
// 可能的值：
//   "正在搜索…"              — web_search 开始前
//   "正在检索记忆…"          — history_backtracking 开始前
//   "正在读取目录…"          — ls_local
//   "正在读取文件…"          — cat_local
//   "正在写入文件…"          — write_local
//   "正在修改文件…"          — fileinput_local
//   "正在执行命令…"          — execute_bash
//   "正在启动子代理…"        — spawn_sub_agent
//   "正在获取子代理结果…"    — get_sub_agent_result
//   "正在读取日历…"          — calendar_list_events
//   "正在创建日历事件…"      — calendar_create_event
//   "正在删除日历事件…"      — calendar_delete_event
// 前端：在 status 事件到来时显示加载指示器；收到下一个 content 或 done 时清除

event: telemetry
data: {
  "platform": "gemini",
  "model_name": "gemini-1.5-pro",
  "input_chars": 15820,
  "output_chars": 950,
  "tool_calls_in_turn": 2
}

event: error
data: {
  "code": "rate_limit",    // 错误代码（见下）
  "message": "API 配额已用尽，请稍后重试"
}
// code 枚举值：
//   "rate_limit"   — API 配额耗尽 / 限速 (429)
//   "server_busy"  — 服务器繁忙 (503 / overloaded)
//   "server_error" — 服务器内部错误 (500)
//   "timeout"      — 请求超时
//   "network"      — 网络连接中断
//   "auth"         — API 密钥无效或未配置
//   "stream_error" — 其他流式传输中断

event: done
data: "[DONE]"                                 // 流结束标志

// 前端处理示例：
//   const es = new EventSource(...)  ← 或用 fetch + ReadableStream
//   es.addEventListener('content', e => fullText += e.data)
//   es.addEventListener('thinking', e => thinkingText += e.data)
//   es.addEventListener('reasoning', e => reasoningSteps.push(e.data))
//   es.addEventListener('telemetry', e => displayUsage(JSON.parse(e.data)))
//   es.addEventListener('done', () => es.close())

20. POST /api/agents/conversations/<pk>/branch/
    用途：从指定 AI 消息处分叉，复制历史消息生成新的独立对话（Branch Here）
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// Request Body（JSON）：
{
  "branch_from_message_id": 102   // 分叉点 AI 消息的 id（必须属于该 conversation 且 role="assistant"）
}

// Response（201 Created）：
{
  "session_id":      15,                        // 新建 AgentSession.id（用于后续 chat 接口）
  "conversation_id": 28,                        // 新建 Conversation.id
  "name":            "Branch from 量子纠缠讨论"  // 自动生成的名称
}

// 分叉规则：
//   - 复制从对话开始到 branch_from_message_id 所在位置的消息（含），最多取末尾 21 条
//   - 确保首条为 user 消息、末条为 assistant 消息
//   - 新对话继承：project / agent_preset / session_type / thinking_level / temperature
//   - 新对话重置：last_summarized_index=0 / active_tool_groups=[] / Proposal 不复制
//   - 附件关联不复制（attachment_ids=[]），消息文本正文完整保留
